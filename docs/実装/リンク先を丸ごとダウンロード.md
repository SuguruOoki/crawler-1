# リンク先を丸ごとダウンロード

## 相対パスを展開する
```python
from urllib.parse import urljoin

base_url = "http://example.com/html/a.html"

print(urljoin(base_url, "b.html"))
# http://example.com/html/b.html
print(urljoin(base_url, "sub/c.html"))
# http://example.com/html/sub/c.html
print(urljoin(base_url, "../index.html"))
# http://example.com/index.html
print(urljoin(base_url, "../img/hoge.png"))
# http://example.com/img/hoge.png
```
